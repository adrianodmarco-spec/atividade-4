var lengthOfLongestSubstring = function(s) { let inicioJanela = 0; let maiorTamanho = 0; // Guarda a última posição encontrada de cada caractere const ultimoIndice = {}; for (let indiceAtual = 0; indiceAtual < s.length; indiceAtual++) { const caractere = s[indiceAtual]; // Se o caractere já apareceu dentro da janela atual if ( ultimoIndice[caractere] !== undefined && ultimoIndice[caractere] >= inicioJanela ) { inicioJanela = ultimoIndice[caractere] + 1; } // Atualiza a posição do caractere ultimoIndice[caractere] = indiceAtual; // Calcula o tamanho atual da substring let tamanhoAtual = indiceAtual - inicioJanela + 1; // Atualiza o maior tamanho encontrado if (tamanhoAtual > maiorTamanho) { maiorTamanho = tamanhoAtual; } } return maiorTamanho; };
